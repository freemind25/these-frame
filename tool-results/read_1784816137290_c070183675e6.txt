    67→// ─── Component ─────────────────────────────────────────────────
    68→export default function Home() {
    69→  const [thesis, setThesis] = useState<ThesisData | null>(null)
    70→  const [loading, setLoading] = useState(true)
    71→  const [activeChapterId, setActiveChapterId] = useState<string>('')
    72→  const [saveStatus, setSaveStatus] = useState<'idle' | 'saving' | 'saved' | 'error'>('idle')
    73→
    74→  // UI state
    75→  const [sidebarOpen, setSidebarOpen] = useState(false)
    76→  const [helpOpen, setHelpOpen] = useState(true)
    77→  const [helpTab, setHelpTab] = useState('guide')
    78→  const [refsOpen, setRefsOpen] = useState(false)
    79→  const [exportOpen, setExportOpen] = useState(false)
    80→  const [resourcesOpen, setResourcesOpen] = useState(false)
    81→  const isMobile = useIsMobile()
    82→
    83→  // AI chat state (in help panel)
    84→  const [aiMessages, setAiMessages] = useState<ChatMsg[]>([])
    85→  const [aiInput, setAiInput] = useState('')
    86→  const [aiLoading, setAiLoading] = useState(false)
    87→  const [aiMode, setAiMode] = useState('scientific-writing')
    88→
    89→  // AI provider state
    90→  const [providerSettingsOpen, setProviderSettingsOpen] = useState(false)
    91→  const [aiProvider, setAiProvider] = useState(() => {
    92→    if (typeof window !== 'undefined') return localStorage.getItem('tf_provider') || 'z-ai'
    93→    return 'z-ai'
    94→  })
    95→  const [aiApiKey, setAiApiKey] = useState(() => {
    96→    if (typeof window !== 'undefined') return localStorage.getItem('tf_apiKey') || ''
    97→    return ''
    98→  })
    99→  const [aiBaseUrl, setAiBaseUrl] = useState(() => {
   100→    if (typeof window !== 'undefined') return localStorage.getItem('tf_baseUrl') || ''
   101→    return ''
   102→  })
   103→  const [aiModel, setAiModel] = useState(() => {
   104→    if (typeof window !== 'undefined') return localStorage.getItem('tf_model') || ''
   105→    return ''
   106→  })
   107→
   108→  const saveProviderSettings = useCallback(() => {
   109→    localStorage.setItem('tf_provider', aiProvider)
   110→    localStorage.setItem('tf_apiKey', aiApiKey)
   111→    localStorage.setItem('tf_baseUrl', aiBaseUrl)
   112→    localStorage.setItem('tf_model', aiModel)
   113→    setProviderSettingsOpen(false)
   114→  }, [aiProvider, aiApiKey, aiBaseUrl, aiModel])
   115→
   116→  const clearProviderSettings = useCallback(() => {
   117→    localStorage.removeItem('tf_provider')
   118→    localStorage.removeItem('tf_apiKey')
   119→    localStorage.removeItem('tf_baseUrl')
   120→    localStorage.removeItem('tf_model')
   121→    setAiProvider('z-ai')
   122→    setAiApiKey('')
   123→    setAiBaseUrl('')
   124→    setAiModel('')
   125→  }, [])
   126→
   127→  // Director state
   128→  const [directorLoading, setDirectorLoading] = useState(false)
   129→  const [directorFeedback, setDirectorFeedback] = useState('')
   130→
   131→  // Auto-save
   132→  const saveTimer = useRef<ReturnType<typeof setTimeout> | null>(null)
   133→  const localContent = useRef<string>('')
   134→
   135→  // Load thesis on mount
   136→  useEffect(() => {
   137→    async function load() {
   138→      try {
   139→        // Seed ensures thesis exists
   140→        await fetch('/api/thesis/seed', { method: 'POST' })
   141→        const res = await fetch('/api/thesis')
   142→        const data = await res.json()
   143→        const thesisData = data.thesis || data
   144→        if (thesisData?.id) {
   145→          setThesis(thesisData)
   146→          if (thesisData.chapters?.length > 0) {
   147→            setActiveChapterId(thesisData.chapters[0].id)
   148→          }
   149→        }
   150→      } catch (err) {
   151→        console.error('Failed to load thesis:', err)
   152→      } finally {
   153→        setLoading(false)
   154→      }
   155→    }
   156→    load()
   157→  }, [])
   158→
   159→  const activeChapter = thesis?.chapters.find(c => c.id === activeChapterId)
   160→  const chapterMeta = CHAPTERS.find(c => c.order === activeChapter?.order)
   161→  const colors = chapterMeta ? CHAPTER_COLORS[chapterMeta.color] : CHAPTER_COLORS.emerald
   162→
   163→  // Total word count
   164→  const totalWords = thesis?.chapters.reduce((sum, c) => sum + c.wordCount, 0) || 0
   165→
   166→  // ─── Auto-save handler ───────────────────────────────────
   167→  const handleContentChange = useCallback((content: string) => {
   168→    if (!activeChapter) return
   169→    localContent.current = content
   170→
   171→    // Optimistic update
   172→    setThesis(prev => prev ? {
   173→      ...prev,
   174→      chapters: prev.chapters.map(c =>
   175→        c.id === activeChapterId ? { ...c, content } : c
   176→      ),
   177→    } : null)
   178→
   179→    // Debounced save
   180→    setSaveStatus('idle')
   181→    if (saveTimer.current) clearTimeout(saveTimer.current)
   182→    saveTimer.current = setTimeout(async () => {
   183→ setSaveStatus('saving')
   184→      try {
   185→        const wordCount = content.split(/\s+/).filter(w => w.length > 0).length
   186→        const res = await fetch(`/api/thesis/chapters/${activeChapterId}`, {
   187→          method: 'PATCH',
   188→          headers: { 'Content-Type': 'application/json' },
   189→          body: JSON.stringify({ content, wordCount, status: wordCount > 0 ? 'in_progress' : 'draft' }),
   190→        })
   191→        if (res.ok) {
   192→          const data = await res.json()
   193→          setThesis(prev => prev ? {
   194→            ...prev,
   195→            chapters: prev.chapters.map(c => c.id === data.chapter.id ? { ...c, ...data.chapter } : c),
   196→          } : null)
   197→          setSaveStatus('saved')
   198→          setTimeout(() => setSaveStatus('idle'), 2000)
   199→        } else {
   200→          setSaveStatus('error')
   201→        }
   202→      } catch {
   203→        setSaveStatus('error')
   204→      }
   205→    }, 2000)
   206→  }, [activeChapter, activeChapterId])
   207→
   208→  // ─── Director submit ─────────────────────────────────────
   209→  const handleDirectorSubmit = useCallback(async () => {
   210→    if (!activeChapter || !chapterMeta) return
   211→    setDirectorLoading(true)
   212→    setDirectorFeedback('')
   213→    try {
   214→      const res = await fetch('/api/directeur', {
   215→        method: 'POST',
   216→        headers: { 'Content-Type': 'application/json' },
   217→        body: JSON.stringify({
   218→          chapitreTitre: `${chapterMeta.number}. ${chapterMeta.title}`,
   219→          chapitreContenu: activeChapter.content,
   220→          probleme: { quoi: 'Contenu du chapitre soumis', comment: 'Évaluation qualitative', pourquoi: 'Validation avant passage au chapitre suivant' },
   221→          hypothese: { texte: 'Chapitre soumis pour évaluation', observation: true, verifiable: true, coherente: true },
   222→          sousDomaineLabel: thesis?.field || 'Non précisé',
   223→          contraintesMethodologiques: '',
   224→        }),
   225→      })
   226→      const data = await res.json()
   227→      if (!res.ok) throw new Error(data.error)
   228→      setDirectorFeedback(data.response)
   229→      // Update chapter status
   230→      await fetch(`/api/thesis/chapters/${activeChapterId}`, {
   231→        method: 'PATCH',
   232→        headers: { 'Content-Type': 'application/json' },
   233→        body: JSON.stringify({ status: 'submitted', directorFeedback: data.response }),
   234→      })
   235→    } catch (err) {
   236→      setDirectorFeedback(err instanceof Error ? err.message : 'Erreur lors de la soumission.')
   237→    } finally {
   238→      setDirectorLoading(false)
   239→    }
   240→  }, [activeChapter, chapterMeta, thesis, activeChapterId])
   241→
   242→  // ─── AI chat ────────────────────────────────────────────
   243→  const handleAiSend = useCallback(async () => {
   244→    if (!aiInput.trim() || aiLoading) return
   245→    const msg = aiInput.trim()
   246→    setAiInput('')
   247→    const newMessages: ChatMsg[] = [...aiMessages, { role: 'user', content: msg }]
   248→    setAiMessages(newMessages)
   249→    setAiLoading(true)
   250→    try {
   251→      const reqBody: Record<string, unknown> = { mode: aiMode, message: msg, temperature: 0.7, maxTokens: 2048, thinking: 'disabled' }
   252→      if (aiProvider !== 'z-ai') {
   253→        reqBody.provider = aiProvider
   254→        reqBody.apiKey = aiApiKey
   255→        reqBody.baseUrl = aiBaseUrl
   256→        reqBody.model = aiModel
   257→      }
   258→      const res = await fetch('/api/ai-writing', {
   259→        method: 'POST',
   260→        headers: { 'Content-Type': 'application/json' },
   261→        body: JSON.stringify(reqBody),
   262→      })
   263→      const data = await res.json()
   264→      if (!res.ok) throw new Error(data.error)
   265→      setAiMessages(prev => [...prev, { role: 'assistant', content: data.response }])
   266→    } catch (err) {
   267→      setAiMessages(prev => [...prev, { role: 'assistant', content: `Erreur : ${err instanceof Error ? err.message : 'inconnue'}` }])
   268→    } finally {
   269→      setAiLoading(false)
   270→    }
   271→  }, [aiInput, aiLoading, aiMessages, aiMode])
   272→
   273→  // ─── Render ─────────────────────────────────────────────
   274→  if (loading) {
   275→    return (
   276→      <div className="h-screen flex items-center justify-center bg-slate-50">
   277→        <div className="text-center space-y-3">
   278→          <Loader2 className="h-8 w-8 text-emerald-500 animate-spin mx-auto" />
   279→          <p className="text-sm text-muted-foreground">Chargement de votre thèse...</p>
   280→        </div>
   281→      </div>
   282→    )
   283→  }
   284→
   285→  if (!thesis) {
   286→    return (
   287→      <div className="h-screen flex items-center justify-center bg-slate-50">
   288→        <div className="text-center space-y-3">
   289→          <AlertTriangle className="h-8 w-8 text-amber-500 mx-auto" />
   290→          <p className="text-sm text-muted-foreground">Impossible de charger la thèse.</p>
   291→          <Button onClick={() => window.location.reload()} variant="outline" size="sm"><RefreshCw className="h-3 w-3 mr-1" />Réessayer</Button>
   292→        </div>
   293→      </div>
   294→    )
   295→  }
   296→
   297→  return (
   298→    <div className="h-screen flex flex-col overflow-hidden bg-slate-50">
   299→      {/* ── MOBILE OVERLAY ── */}
   300→      {sidebarOpen && isMobile && (
   301→        <div className="fixed inset-0 bg-black/50 backdrop-blur-sm z-30" onClick={() => setSidebarOpen(false)} />
   302→      )}
   303→
   304→      <div className="flex flex-1 min-h-0">
   305→        {/* ═══ LEFT SIDEBAR ═══ */}
   306→        <aside className={cn(
   307→          'bg-slate-900 border-r border-slate-800 flex flex-col z-40 shrink-0 transition-all duration-300 shadow-2xl',
   308→          'w-64',
   309→          isMobile && !sidebarOpen && 'fixed -translate-x-full',
   310→          isMobile && sidebarOpen && 'fixed translate-x-0',
   311→          !isMobile && 'relative',
   312→        )}>
   313→          {/* Brand */}
   314→          <div className="flex items-center gap-3 px-3 py-4 mb-2">
   315→            <div className="w-9 h-9 rounded-xl bg-gradient-to-tr from-emerald-600 to-emerald-400 flex items-center justify-center text-white shadow-[0_0_16px_rgba(16,185,129,0.3)] shrink-0">
   316→              <GraduationCap className="h-4 w-4" />
   317→            </div>
   318→            <div className="overflow-hidden">
   319→              <h1 className="font-bold text-sm text-white tracking-tight leading-tight">ThesisFrame</h1>
   320→              <p className="text-[10px] text-emerald-400 font-medium">{thesis.title}</p>
   321→            </div>
   322→            {isMobile && (
   323→              <button onClick={() => setSidebarOpen(false)} className="ml-auto text-slate-400 hover:text-white p-1"><X className="h-4 w-4" /></button>
   324→            )}
   325→          </div>
   326→
   327→          {/* Progress */}
   328→          <div className="px-3 pb-3">
   329→            <div className="flex items-center justify-between mb-1">
   330→              <span className="text-[10px] text-slate-500 font-medium">PROGRESSION</span>
   331→              <span className="text-[10px] text-emerald-400 font-bold">{totalWords.toLocaleString()} mots</span>
   332→            </div>
   333→            <div className="h-1.5 bg-slate-800 rounded-full overflow-hidden">
   334→              <div className="h-full bg-gradient-to-r from-emerald-500 to-emerald-400 rounded-full transition-all duration-500" style={{ width: `${Math.min(100, (totalWords / 80000) * 100)}%` }} />
   335→            </div>
   336→          </div>
   337→
   338→          <Separator className="bg-slate-800" />
   339→
   340→          {/* Chapters list */}
   341→          <div className="px-3 pt-2 pb-1">
   342→            <span className="text-[10px] font-bold uppercase tracking-wider text-slate-500">Chapitres</span>
   343→          </div>
   344→          <nav className="flex-1 overflow-y-auto space-y-0.5 px-2">
   345→            {thesis.chapters.map((ch) => {
   346→              const meta = CHAPTERS.find(m => m.order === ch.order)
   347→              const isActive = ch.id === activeChapterId
   348→              const Icon = (meta?.icon && ICON_MAP[meta.icon]) || FileText
   349→              const chColors = meta ? CHAPTER_COLORS[meta.color] : CHAPTER_COLORS.emerald
   350→              return (
   351→                <button
   352→                  key={ch.id}
   353→                  onClick={() => { setActiveChapterId(ch.id); setSidebarOpen(false) }}
   354→                  className={cn(
   355→                    'w-full p-2.5 flex items-start gap-2.5 rounded-xl text-left transition-all duration-200 group relative overflow-hidden',
   356→                    isActive
   357→                      ? 'bg-gradient-to-r from-emerald-900/80 to-slate-900 text-white border border-emerald-500/40'
   358→                      : 'text-slate-400 hover:text-white hover:bg-slate-800/80 border border-transparent',
   359→                  )}
   360→                >
   361→                  <span className="absolute inset-0 bg-gradient-to-r from-transparent via-emerald-500/10 to-transparent -translate-x-full group-hover:translate-x-full transition-transform duration-700" />
   362→                  <div className={cn('w-7 h-7 rounded-lg flex items-center justify-center shrink-0 mt-0.5 transition-colors',
   363→                    isActive ? chColors.bg.replace('bg-', 'bg-').replace('/500', '/500/20') : 'bg-slate-800 group-hover:bg-slate-700',
   364→                  )}>
   365→                    <Icon className={cn('h-3.5 w-3.5', isActive ? 'text-emerald-400' : 'text-slate-400 group-hover:text-emerald-400')} />
   366→                  </div>
   367→                  <div className="flex-1 min-w-0">
   368→                    <div className="flex items-center gap-1.5">
   369→                      <span className={cn('text-[10px] font-bold', isActive ? 'text-emerald-400' : 'text-slate-600')}>{ch.number}</span>
   370→                      <span className={cn('text-[11px] font-semibold leading-tight truncate', isActive && 'text-white')}>{ch.title}</span>
   371→                    </div>
   372→                    <div className="flex items-center gap-1.5 mt-0.5">
   373→                      <div className={cn('h-1.5 w-1.5 rounded-full', STATUS_COLORS[ch.status] || STATUS_COLORS.draft)} />
   374→                      <span className="text-[9px] text-slate-600">{ch.wordCount.toLocaleString()} mots</span>
   375→                    </div>
   376→                  </div>
   377→                </button>
   378→              )
   379→            })}
   380→          </nav>
   381→
   382→          <Separator className="bg-slate-800" />
   383→
   384→          {/* Tools */}
   385→          <div className="px-3 pt-2 pb-1">
   386→            <span className="text-[10px] font-bold uppercase tracking-wider text-slate-500">Outils</span>
   387→          </div>
   388→          <div className="px-2 pb-3 space-y-0.5">
   389→            <button onClick={() => setRefsOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   390→              <Library className="h-3.5 w-3.5" /><span>Références biblio.</span>
   391→            </button>
   392→            <button onClick={() => setResourcesOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   393→              <BookOpen className="h-3.5 w-3.5" /><span>Guide rédaction</span>
   394→            </button>
   395→            <button onClick={() => setExportOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   396→              <Download className="h-3.5 w-3.5" /><span>Export PDF</span>
   397→            </button>
   398→          </div>
   399→
   400→          {/* User */}
   401→          <div className="p-3 border-t border-slate-800">
   402→            <div className="p-2 rounded-xl bg-slate-800/50 border border-slate-700/40 flex items-center gap-2">
   403→              <div className="w-7 h-7 rounded-full bg-gradient-to-br from-emerald-500 to-emerald-700 flex items-center justify-center text-white font-bold text-[10px] ring-2 ring-emerald-500/30 shrink-0">DR</div>
   404→              <div className="overflow-hidden flex-1">
   405→                <div className="text-[10px] font-medium text-slate-200 truncate">{thesis.author}</div>
   406→                <div className="text-[9px] text-slate-500 truncate">{thesis.university}</div>
   407→              </div>
   408→            </div>
   409→          </div>
   410→        </aside>
   411→
   412→        {/* ═══ MAIN AREA ═══ */}
   413→        <div className="flex-1 flex flex-col min-w-0">
   414→          {/* Chapter header bar */}
   415→          <header className="bg-white/90 backdrop-blur-md border-b border-slate-200 px-4 sm:px-6 py-2.5 flex items-center gap-3 shrink-0 z-20">
   416→            {isMobile && (
   417→              <button onClick={() => setSidebarOpen(true)} className="p-1.5 text-slate-600 hover:text-slate-900 hover:bg-slate-100 rounded-lg">
   418→                <Menu className="h-4 w-4" />
   419→              </button>
   420→            )}
   421→            {chapterMeta && (
   422→              <>
   423→                <div className={cn('p-1.5 rounded-lg flex items-center justify-center', colors.light, colors.text)}>
   424→                  {createElement(ICON_MAP[chapterMeta.icon] || FileText, { className: 'h-3.5 w-3.5' })}
   425→                </div>
   426→                <div className="flex-1 min-w-0">
   427→                  <div className="flex items-center gap-2">
   428→                    <h2 className="text-sm sm:text-base font-bold text-slate-900 tracking-tight truncate">
   429→                      Chapitre {chapterMeta.number}. {chapterMeta.title}
   430→                    </h2>
   431→                  </div>
   432→                  <p className="text-[10px] sm:text-xs text-slate-500 truncate">{chapterMeta.description}</p>
   433→                </div>
   434→                <div className="flex items-center gap-2 shrink-0">
   435→                  <Badge variant="outline" className="text-[10px] gap-1">
   436→                    <span className={cn('h-1.5 w-1.5 rounded-full', STATUS_COLORS[activeChapter?.status || 'draft'])} />
   437→                    {activeChapter?.status || 'brouillon'}
   438→                  </Badge>
   439→                  <Badge variant="secondary" className="text-[10px] hidden sm:inline-flex">
   440→                    {activeChapter?.wordCount?.toLocaleString() || 0} mots
   441→                  </Badge>
   442→                  {saveStatus === 'saving' && <Loader2 className="h-3 w-3 text-slate-400 animate-spin" />}
   443→                  {saveStatus === 'saved' && <Check className="h-3 w-3 text-emerald-500" />}
   444→                </div>
   445→              </>
   446→            )}
   447→            <button
   448→              onClick={() => setHelpOpen(!helpOpen)}
   449→              className="p-1.5 text-slate-500 hover:text-slate-800 hover:bg-slate-100 rounded-lg transition-colors"
   450→              title={helpOpen ? 'Fermer le panneau' : 'Ouvrir l\'aide'}
   451→            >
   452→              {helpOpen ? <PanelRightClose className="h-4 w-4" /> : <PanelRightOpen className="h-4 w-4" />}
   453→            </button>
   454→          </header>

   456→          {/* Editor + Help panel */}
