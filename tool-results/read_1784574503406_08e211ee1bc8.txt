   395→// ─── ReferenceRow ────────────────────────────────────────────────
   396→function ReferenceRow({ item, onDelete }: { item: RefItem; onDelete: (id: string) => void }) {
   397→  const [expanded, setExpanded] = useState(false)
   398→  const [copied, setCopied] = useState(false)
   399→
   400→  const copyBib = () => {
   401→    const key = item.citationKey || `ref_${item.id.slice(0, 6)}`
   402→    const authorList = item.authors ? item.authors.split(';').map(a => a.trim()).join(' and ') : 'Unknown'
   403→    let bib = ''
   404→    if (item.type === 'book') {
   405→      bib = `@book{${key},\n  author = {${authorList}},\n  title = {${item.title}},\n  year = {${item.year || ''}},\n  publisher = {${item.journal || ''}}\n}`
   406→    } else {
   407→      bib = `@article{${key},\n  author = {${authorList}},\n  title = {${item.title}},\n  journal = {${item.journal || ''}},\n  year = {${item.year || ''}},\n  volume = {${item.volume || ''}},\n  pages = {${item.pages || ''}},\n  doi = {${item.doi || ''}}\n}`
   408→    }
   409→    navigator.clipboard.writeText(bib)
   410→    setCopied(true)
   411→    setTimeout(() => setCopied(false), 2000)
   412→  }
   413→
   414→  const typeLabels: Record<string, string> = {
   415→    article: 'Article',
   416→    book: 'Livre',
   417→    inproceedings: 'Conférence',
   418→    thesis: 'Thèse',
   419→    incollection: 'Chapitre',
   420→    web: 'Web',
   421→  }
   422→
   423→  return (
   424→    <div className="rounded-lg border hover:bg-muted/30 transition-colors">
   425→      <div
   426→        className="flex items-start gap-3 p-3 cursor-pointer select-none"
   427→        onClick={() => setExpanded(!expanded)}
   428→      >
   429→        <ChevronRight className={`h-4 w-4 text-muted-foreground shrink-0 mt-1 transition-transform ${expanded ? 'rotate-90' : ''}`} />
   430→        <div className="flex-1 min-w-0">
   431→          <div className="flex items-center gap-2 flex-wrap">
   432→            <span className="text-sm font-medium truncate">{item.title}</span>
   433→            <Badge variant="secondary" className="text-xs shrink-0">{typeLabels[item.type] || item.type}</Badge>
   434→            {item.source === 'mendeley' && (
   435→              <Badge variant="outline" className="text-xs shrink-0 text-amber-600 border-amber-300 bg-amber-50">Mendeley</Badge>
   436→            )}
   437→          </div>
   438→          <p className="text-xs text-muted-foreground mt-0.5 truncate">{item.authors} ({item.year || '—'})</p>
   439→          {item.journal && <p className="text-xs text-muted-foreground truncate">{item.journal}{item.volume ? `, ${item.volume}` : ''}{item.pages ? `, pp. ${item.pages}` : ''}</p>}
   440→        </div>
   441→        <div className="flex items-center gap-1 shrink-0">
   442→          <Button variant="ghost" size="icon" className="h-7 w-7" onClick={(e) => { e.stopPropagation(); copyBib() }} title="Copier BibTeX">
   443→            <Copy className="h-3.5 w-3.5" />
   444→          </Button>
   445→          <Button variant="ghost" size="icon" className="h-7 w-7 text-red-500 hover:text-red-600" onClick={(e) => { e.stopPropagation(); onDelete(item.id) }} title="Supprimer">
   446→            <Trash2 className="h-3.5 w-3.5" />
   447→          </Button>
   448→        </div>
   449→      </div>
   450→      {expanded && (
   451→        <div className="px-3 pb-3 pl-10 space-y-2 border-t pt-2 ml-7">
   452→          <div className="grid grid-cols-2 sm:grid-cols-4 gap-2 text-xs">
   453→            <div><span className="text-muted-foreground">Clé :</span> <span className="font-mono">{item.citationKey || '—'}</span></div>
   454→            <div><span className="text-muted-foreground">DOI :</span> {item.doi || '—'}</div>
   455→            <div><span className="text-muted-foreground">Volume :</span> {item.volume || '—'}</div>
   456→            <div><span className="text-muted-foreground">N° :</span> {item.number || '—'}</div>
   457→          </div>
   458→          {item.abstract && (
   459→            <div>
   460→              <span className="text-xs font-semibold text-muted-foreground">Résumé</span>
   461→              <p className="text-xs text-muted-foreground mt-1 bg-muted/50 rounded p-2 max-h-32 overflow-y-auto">{item.abstract}</p>
   462→            </div>
   463→          )}
   464→          {item.notes && (
   465→            <div>
   466→              <span className="text-xs font-semibold text-muted-foreground">Notes</span>
   467→              <p className="text-xs mt-1 italic">{item.notes}</p>
   468→            </div>
   469→          )}
   470→          {item.tags && (
   471→            <div className="flex items-center gap-1.5 flex-wrap">
   472→              {item.tags.split(',').map((tag, i) => (
   473→                <Badge key={i} variant="outline" className="text-xs">{tag.trim()}</Badge>
   474→              ))}
   475→            </div>
   476→          )}
   477→          {copied && (
   478→            <p className="text-xs text-emerald-600 flex items-center gap-1">
   479→              <CheckCircle2 className="h-3 w-3" /> BibTeX copié dans le presse-papier
   480→            </p>
   481→          )}
   482→        </div>
   483→      )}
   484→    </div>
   485→  )
   486→}

   488→// ─── MendeleyImportPanel ─────────────────────────────────────────
   489→function MendeleyImportPanel({ connected, onImport }: { connected: boolean; onImport: (docs: unknown[]) => void }) {
   490→  const [documents, setDocuments] = useState<unknown[]>([])
   491→  const [loading, setLoading] = useState(false)
   492→  const [selected, setSelected] = useState<Set<string>>(new Set())
   493→  const [search, setSearch] = useState('')
   494→
   495→  const fetchDocs = useCallback(async () => {
   496→    if (!connected) return
   497→    setLoading(true)
   498→    try {
   499→      const res = await fetch('/api/mendeley/documents?limit=50')
   500→      if (res.ok) {
   501→        const data = await res.json()
   502→        setDocuments(Array.isArray(data) ? data : [])
   503→      }
   504→    } catch { /* ignore */ }
   505→    setLoading(false)
   506→  }, [connected])
   507→
   508→  // eslint-disable-next-line react-hooks/set-state-in-effect
   509→  useEffect(() => { if (connected) { fetchDocs(); } }, [connected, fetchDocs])
   510→

   925→    setEditingField(null)
