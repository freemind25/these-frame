   115→const STORAGE_KEY = 'thesisframe-ai-provider'
   116→
   117→function loadProviderConfig(): ProviderConfig {
   118→  if (typeof window === 'undefined') return { provider: 'z-ai', apiKey: '', model: 'default', customBaseUrl: '', customModel: '' }
   119→  try {
   120→    const saved = localStorage.getItem(STORAGE_KEY)
   121→    if (saved) return JSON.parse(saved)
   122→  } catch { /* ignore */ }
   123→  return { provider: 'z-ai', apiKey: '', model: 'default', customBaseUrl: '', customModel: '' }
   124→}
   125→
   126→function saveProviderConfig(cfg: ProviderConfig) {
   127→  if (typeof window !== 'undefined') {
   128→    localStorage.setItem(STORAGE_KEY, JSON.stringify(cfg))
   129→  }
   130→}

   225→  function updateProvider<K extends keyof ProviderConfig>(key: K, value: ProviderConfig[K]) {
   226→    const updated = { ...providerConfig, [key]: value }
   227→    if (key === 'provider') {
   228→      // Reset model to first when provider changes
   229→      updated.model = PROVIDER_MODELS[value][0].value
   230→    }
   231→    setProviderConfig(updated)
   232→    saveProviderConfig(updated)
   233→  }

   235→  function getEffectiveModel(): string {
   236→    if (providerConfig.provider === 'z-ai') return 'default'
   237→    if (providerConfig.provider === 'custom') return providerConfig.customModel || 'default'
   238→    return providerConfig.model
   239→  }
   240→
   241→  function getEffectiveBaseUrl(): string {
   242→    if (providerConfig.provider === 'z-ai') return ''
   243→    if (providerConfig.provider === 'custom') return providerConfig.customBaseUrl || ''
   244→    return PROVIDER_BASE_URLS[providerConfig.provider]
   245→  }
   246→
   247→  async function handleSend() {
   248→    if (!input.trim() || isLoading) return
   249→    setError('')
   250→    const userMsg = input.trim()
   251→    setInput('')
   252→
   253→    // Validate provider config
   254→    if (needsApiKey && !providerConfig.apiKey.trim()) {
   255→      setError(`Clé API requise pour ${activeProvider?.label}. Ouvrez la Configuration IA.`)
   256→      setShowConfig(true)
   257→      return
   258→    }
   259→    if (providerConfig.provider === 'custom' && !providerConfig.customBaseUrl.trim()) {
   260→      setError('URL de base requise pour le fournisseur personnalisé.')
   261→      setShowConfig(true)
   262→      return
   263→    }
   264→    if (providerConfig.provider === 'custom' && !providerConfig.customModel.trim()) {
   265→      setError('Nom du modèle requis pour le fournisseur personnalisé.')
   266→      setShowConfig(true)
   267→      return
   268→    }
   269→
   270→    const newMessages: ChatMessage[] = [...messages, { role: 'user', content: userMsg }]
   271→    setMessages(newMessages)
   272→    setIsLoading(true)
   273→
   274→    try {
   275→      const body: Record<string, unknown> = {
   276→        mode: selectedMode,
   277→        message: userMsg,
   278→        sessionId: sessionId || undefined,
   279→        temperature,
   280→        maxTokens,
   281→        thinking: thinkingEnabled ? 'enabled' : 'disabled',
   282→      }
   283→
   284→      // Include provider info for non-z-ai providers
   285→      if (providerConfig.provider !== 'z-ai') {
   286→        body.provider = providerConfig.provider
   287→        body.apiKey = providerConfig.apiKey
   288→        body.model = getEffectiveModel()
   289→        body.baseUrl = getEffectiveBaseUrl()
   290→      }
   291→
   292→      const res = await fetch('/api/ai-writing', {
   293→        method: 'POST',
   294→        headers: { 'Content-Type': 'application/json' },
   295→        body: JSON.stringify(body),
   296→      })
   297→      const data = await res.json()
   298→      if (!res.ok) throw new Error(data.error || 'Erreur serveur')
   299→      if (data.sessionId && !sessionId) setSessionId(data.sessionId)
   300→      setMessages(prev => [...prev, { role: 'assistant', content: data.response }])
   301→    } catch (err) {
   302→      setError(err instanceof Error ? err.message : 'Erreur inconnue')
   303→    } finally {
   304→      setIsLoading(false)
   305→    }
   306→  }
   307→

   313→    setSessionId('')
   314→    setError('')
   315→  }
