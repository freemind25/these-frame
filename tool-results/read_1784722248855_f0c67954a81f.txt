   436→// ─── ReferenceRow ────────────────────────────────────────────────
   437→function ReferenceRow({ item, onDelete }: { item: RefItem; onDelete: (id: string) => void }) {
   438→  const [expanded, setExpanded] = useState(false)
   439→  const [copied, setCopied] = useState(false)
   440→
   441→  const copyBib = () => {
   442→    const key = item.citationKey || `ref_${item.id.slice(0, 6)}`
   443→    const authorList = item.authors ? item.authors.split(';').map(a => a.trim()).join(' and ') : 'Unknown'
   444→    let bib = ''
   445→    if (item.type === 'book') {
   446→      bib = `@book{${key},\n  author = {${authorList}},\n  title = {${item.title}},\n  year = {${item.year || ''}},\n  publisher = {${item.journal || ''}}\n}`
   447→    } else {
   448→      bib = `@article{${key},\n  author = {${authorList}},\n  title = {${item.title}},\n  journal = {${item.journal || ''}},\n  year = {${item.year || ''}},\n  volume = {${item.volume || ''}},\n  pages = {${item.pages || ''}},\n  doi = {${item.doi || ''}}\n}`
   449→    }
   450→    navigator.clipboard.writeText(bib)
   451→    setCopied(true)
   452→    setTimeout(() => setCopied(false), 2000)
   453→  }
   454→
   455→  const typeLabels: Record<string, string> = {
   456→    article: 'Article',
   457→    book: 'Livre',
   458→    inproceedings: 'Conférence',
   459→    thesis: 'Thèse',
   460→    incollection: 'Chapitre',
   461→    web: 'Web',
   462→  }
   463→
   464→  return (
   465→    <div className="rounded-lg border hover:bg-muted/30 transition-colors">
   466→      <div
   467→        className="flex items-start gap-3 p-3 cursor-pointer select-none"
   468→        onClick={() => setExpanded(!expanded)}
   469→      >
   470→        <ChevronRight className={`h-4 w-4 text-muted-foreground shrink-0 mt-1 transition-transform ${expanded ? 'rotate-90' : ''}`} />
   471→        <div className="flex-1 min-w-0">
   472→          <div className="flex items-center gap-2 flex-wrap">
   473→            <span className="text-sm font-medium truncate">{item.title}</span>
   474→            <Badge variant="secondary" className="text-xs shrink-0">{typeLabels[item.type] || item.type}</Badge>
   475→            {item.source === 'mendeley' && (
   476→              <Badge variant="outline" className="text-xs shrink-0 text-amber-600 border-amber-300 bg-amber-50">Mendeley</Badge>
   477→            )}
   478→          </div>
   479→          <p className="text-xs text-muted-foreground mt-0.5 truncate">{item.authors} ({item.year || '—'})</p>
   480→          {item.journal && <p className="text-xs text-muted-foreground truncate">{item.journal}{item.volume ? `, ${item.volume}` : ''}{item.pages ? `, pp. ${item.pages}` : ''}</p>}
   481→        </div>
   482→        <div className="flex items-center gap-1 shrink-0">
   483→          <Button variant="ghost" size="icon" className="h-7 w-7" onClick={(e) => { e.stopPropagation(); copyBib() }} title="Copier BibTeX">
   484→            <Copy className="h-3.5 w-3.5" />
   485→          </Button>
   486→          <Button variant="ghost" size="icon" className="h-7 w-7 text-red-500 hover:text-red-600" onClick={(e) => { e.stopPropagation(); onDelete(item.id) }} title="Supprimer">
   487→            <Trash2 className="h-3.5 w-3.5" />
   488→          </Button>
   489→        </div>
   490→      </div>
   491→      {expanded && (
   492→        <div className="px-3 pb-3 pl-10 space-y-2 border-t pt-2 ml-7">
   493→          <div className="grid grid-cols-2 sm:grid-cols-4 gap-2 text-xs">
   494→            <div><span className="text-muted-foreground">Clé :</span> <span className="font-mono">{item.citationKey || '—'}</span></div>
   495→            <div><span className="text-muted-foreground">DOI :</span> {item.doi || '—'}</div>
   496→            <div><span className="text-muted-foreground">Volume :</span> {item.volume || '—'}</div>
   497→            <div><span className="text-muted-foreground">N° :</span> {item.number || '—'}</div>
   498→          </div>
   499→          {item.abstract && (
   500→            <div>
   501→              <span className="text-xs font-semibold text-muted-foreground">Résumé</span>
   502→              <p className="text-xs text-muted-foreground mt-1 bg-muted/50 rounded p-2 max-h-32 overflow-y-auto">{item.abstract}</p>
   503→            </div>
   504→          )}
   505→          {item.notes && (
   506→            <div>
   507→              <span className="text-xs font-semibold text-muted-foreground">Notes</span>
   508→              <p className="text-xs mt-1 italic">{item.notes}</p>
   509→            </div>
   510→          )}
   511→          {item.tags && (
   512→            <div className="flex items-center gap-1.5 flex-wrap">
   513→              {item.tags.split(',').map((tag, i) => (
   514→                <Badge key={i} variant="outline" className="text-xs">{tag.trim()}</Badge>
   515→              ))}
   516→            </div>
   517→          )}
   518→          {copied && (
   519→            <p className="text-xs text-emerald-600 flex items-center gap-1">
   520→              <CheckCircle2 className="h-3 w-3" /> BibTeX copié dans le presse-papier

   968→
   969→function ThesisPlanTab() {
   970→  const [chapters, setChapters] = useState<Chapter[]>(DEFAULT_CHAPTERS)
   971→  const [editingId, setEditingId] = useState<string | null>(null)
   972→  const [editingField, setEditingField] = useState<'words' | 'pages' | 'title' | 'notes' | null>(null)
   973→  const [editValue, setEditValue] = useState('')
   974→
   975→  const totalWords = chapters.reduce((a, c) => a + c.words, 0)
   976→  const totalPages = chapters.reduce((a, c) => a + c.pages, 0)
   977→  const startedCount = chapters.filter(c => c.words > 0 || c.pages > 0).length
   978→
   979→  function startEdit(id: string, field: 'words' | 'pages' | 'title' | 'notes', currentValue: string | number) {
   980→    setEditingId(id)
   981→    setEditingField(field)
   982→    setEditValue(String(currentValue))
   983→  }
   984→
   985→  function commitEdit() {
   986→    if (!editingId || !editingField) return
   987→    setChapters(prev =>
   988→      prev.map(ch => {
   989→        if (ch.id !== editingId) return ch
   990→        const val = editingField === 'title' || editingField === 'notes'
   991→          ? editValue
   992→          : Math.max(0, parseInt(editValue) || 0)
   993→        return { ...ch, [editingField]: val }
   994→      })
   995→    )
   996→    setEditingId(null)
   997→    setEditingField(null)
   998→    setEditValue('')
   999→  }
  1000→
  1001→  function cancelEdit() {
  1002→    setEditingId(null)
  1003→    setEditingField(null)
  1004→    setEditValue('')
  1005→  }
  1006→
  1007→  function toggleExpand(id: string) {
  1008→    setChapters(prev => prev.map(ch => ch.id === id ? { ...ch, expanded: !ch.expanded } : ch))
  1009→  }
  1010→
  1011→  return (
  1012→    <div className="space-y-6">
  1013→      {/* Header stats */}
  1014→      <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
  1015→        <Card>
  1016→          <CardHeader className="pb-2">
  1017→            <CardDescription>Mots rédigés</CardDescription>
  1018→            <CardTitle className="text-3xl font-bold tabular-nums">
  1019→              {totalWords.toLocaleString()}
  1020→            </CardTitle>
  1021→          </CardHeader>
  1022→          <CardContent>
  1023→            <p className="text-xs text-muted-foreground mt-1">
  1024→              {startedCount} / {chapters.length} chapitres commencés
  1025→            </p>
  1026→          </CardContent>
  1027→        </Card>
