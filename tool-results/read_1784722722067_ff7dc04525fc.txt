   439→// ─── ReferenceRow ────────────────────────────────────────────────
   440→function ReferenceRow({ item, onDelete }: { item: RefItem; onDelete: (id: string) => void }) {
   441→  const [expanded, setExpanded] = useState(false)
   442→  const [copied, setCopied] = useState(false)
   443→
   444→  const copyBib = () => {
   445→    const key = item.citationKey || `ref_${item.id.slice(0, 6)}`
   446→    const authorList = item.authors ? item.authors.split(';').map(a => a.trim()).join(' and ') : 'Unknown'
   447→    let bib = ''
   448→    if (item.type === 'book') {
   449→      bib = `@book{${key},\n  author = {${authorList}},\n  title = {${item.title}},\n  year = {${item.year || ''}},\n  publisher = {${item.journal || ''}}\n}`
   450→    } else {
   451→      bib = `@article{${key},\n  author = {${authorList}},\n  title = {${item.title}},\n  journal = {${item.journal || ''}},\n  year = {${item.year || ''}},\n  volume = {${item.volume || ''}},\n  pages = {${item.pages || ''}},\n  doi = {${item.doi || ''}}\n}`
   452→    }
   453→    navigator.clipboard.writeText(bib)
   454→    setCopied(true)
   455→    setTimeout(() => setCopied(false), 2000)
   456→  }
   457→
   458→  const typeLabels: Record<string, string> = {
   459→    article: 'Article',
   460→    book: 'Livre',
   461→    inproceedings: 'Conférence',
   462→    thesis: 'Thèse',
   463→    incollection: 'Chapitre',
   464→    web: 'Web',
   465→  }
   466→
   467→  return (
   468→    <div className="rounded-lg border hover:bg-muted/30 transition-colors">
   469→      <div
   470→        className="flex items-start gap-3 p-3 cursor-pointer select-none"
   471→        onClick={() => setExpanded(!expanded)}
   472→      >
   473→        <ChevronRight className={`h-4 w-4 text-muted-foreground shrink-0 mt-1 transition-transform ${expanded ? 'rotate-90' : ''}`} />
   474→        <div className="flex-1 min-w-0">
   475→          <div className="flex items-center gap-2 flex-wrap">
   476→            <span className="text-sm font-medium truncate">{item.title}</span>
   477→            <Badge variant="secondary" className="text-xs shrink-0">{typeLabels[item.type] || item.type}</Badge>
   478→            {item.source === 'mendeley' && (
   479→              <Badge variant="outline" className="text-xs shrink-0 text-amber-600 border-amber-300 bg-amber-50">Mendeley</Badge>
   480→            )}
   481→          </div>
   482→          <p className="text-xs text-muted-foreground mt-0.5 truncate">{item.authors} ({item.year || '—'})</p>
   483→          {item.journal && <p className="text-xs text-muted-foreground truncate">{item.journal}{item.volume ? `, ${item.volume}` : ''}{item.pages ? `, pp. ${item.pages}` : ''}</p>}
   484→        </div>
   485→        <div className="flex items-center gap-1 shrink-0">
   486→          <Button variant="ghost" size="icon" className="h-7 w-7" onClick={(e) => { e.stopPropagation(); copyBib() }} title="Copier BibTeX">
   487→            <Copy className="h-3.5 w-3.5" />
   488→          </Button>
   489→          <Button variant="ghost" size="icon" className="h-7 w-7 text-red-500 hover:text-red-600" onClick={(e) => { e.stopPropagation(); onDelete(item.id) }} title="Supprimer">
   490→            <Trash2 className="h-3.5 w-3.5" />
   491→          </Button>
   492→        </div>
   493→      </div>
   494→      {expanded && (
   495→        <div className="px-3 pb-3 pl-10 space-y-2 border-t pt-2 ml-7">
   496→          <div className="grid grid-cols-2 sm:grid-cols-4 gap-2 text-xs">
   497→            <div><span className="text-muted-foreground">Clé :</span> <span className="font-mono">{item.citationKey || '—'}</span></div>
   498→            <div><span className="text-muted-foreground">DOI :</span> {item.doi || '—'}</div>
   499→            <div><span className="text-muted-foreground">Volume :</span> {item.volume || '—'}</div>
   500→            <div><span className="text-muted-foreground">N° :</span> {item.number || '—'}</div>
   501→          </div>
   502→          {item.abstract && (
   503→            <div>
   504→              <span className="text-xs font-semibold text-muted-foreground">Résumé</span>
   505→              <p className="text-xs text-muted-foreground mt-1 bg-muted/50 rounded p-2 max-h-32 overflow-y-auto">{item.abstract}</p>
   506→            </div>
   507→          )}
   508→          {item.notes && (
   509→            <div>
   510→              <span className="text-xs font-semibold text-muted-foreground">Notes</span>
   511→              <p className="text-xs mt-1 italic">{item.notes}</p>
   512→            </div>
   513→          )}
   514→          {item.tags && (
   515→            <div className="flex items-center gap-1.5 flex-wrap">
   516→              {item.tags.split(',').map((tag, i) => (
   517→                <Badge key={i} variant="outline" className="text-xs">{tag.trim()}</Badge>
   518→              ))}
   519→            </div>
   520→          )}

   972→function ThesisPlanTab() {
   973→  const [chapters, setChapters] = useState<Chapter[]>(DEFAULT_CHAPTERS)
   974→  const [editingId, setEditingId] = useState<string | null>(null)
   975→  const [editingField, setEditingField] = useState<'words' | 'pages' | 'title' | 'notes' | null>(null)
   976→  const [editValue, setEditValue] = useState('')
   977→
   978→  const totalWords = chapters.reduce((a, c) => a + c.words, 0)
   979→  const totalPages = chapters.reduce((a, c) => a + c.pages, 0)
   980→  const startedCount = chapters.filter(c => c.words > 0 || c.pages > 0).length
   981→
   982→  function startEdit(id: string, field: 'words' | 'pages' | 'title' | 'notes', currentValue: string | number) {
   983→    setEditingId(id)
   984→    setEditingField(field)
   985→    setEditValue(String(currentValue))
   986→  }
   987→
   988→  function commitEdit() {
   989→    if (!editingId || !editingField) return
   990→    setChapters(prev =>
   991→      prev.map(ch => {
   992→        if (ch.id !== editingId) return ch
   993→        const val = editingField === 'title' || editingField === 'notes'
   994→          ? editValue
   995→          : Math.max(0, parseInt(editValue) || 0)
   996→        return { ...ch, [editingField]: val }
   997→      })
   998→    )
   999→    setEditingId(null)
  1000→    setEditingField(null)
  1001→    setEditValue('')
  1002→  }
  1003→
  1004→  function cancelEdit() {
  1005→    setEditingId(null)
  1006→    setEditingField(null)
  1007→    setEditValue('')
  1008→  }
  1009→
  1010→  function toggleExpand(id: string) {
  1011→    setChapters(prev => prev.map(ch => ch.id === id ? { ...ch, expanded: !ch.expanded } : ch))
  1012→  }
  1013→
  1014→  return (
  1015→    <Tabs defaultValue="planner" className="w-full">
  1016→      <TabsList className="w-full grid grid-cols-3 mb-4">
  1017→        <TabsTrigger value="planner" className="text-xs sm:text-sm flex items-center gap-1.5 justify-center">
  1018→          <Layout className="h-3.5 w-3.5" />
  1019→          Planificateur
