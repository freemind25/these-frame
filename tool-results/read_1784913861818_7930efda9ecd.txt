    71→// ─── Component ─────────────────────────────────────────────────
    72→export default function Home() {
    73→  const [thesis, setThesis] = useState<ThesisData | null>(null)
    74→  const [loading, setLoading] = useState(true)
    75→  const [activeChapterId, setActiveChapterId] = useState<string>('')
    76→  const [saveStatus, setSaveStatus] = useState<'idle' | 'saving' | 'saved' | 'error'>('idle')
    77→
    78→  // UI state
    79→  const [sidebarOpen, setSidebarOpen] = useState(false)
    80→  const [helpOpen, setHelpOpen] = useState(true)
    81→  const [helpTab, setHelpTab] = useState('guide')
    82→  const [refsOpen, setRefsOpen] = useState(false)
    83→  const [exportOpen, setExportOpen] = useState(false)
    84→  const [resourcesOpen, setResourcesOpen] = useState(false)
    85→  const [literatureOpen, setLiteratureOpen] = useState(false)
    86→  const [balanceOpen, setBalanceOpen] = useState(false)
    87→  const [cloudDriveOpen, setCloudDriveOpen] = useState(false)
    88→  const [journalFinderOpen, setJournalFinderOpen] = useState(false)
    89→  const isMobile = useIsMobile()
    90→
    91→  // AI chat state (in help panel)
    92→  const [aiMessages, setAiMessages] = useState<ChatMsg[]>([])
    93→  const [aiInput, setAiInput] = useState('')
    94→  const [aiLoading, setAiLoading] = useState(false)
    95→  const [aiMode, setAiMode] = useState('scientific-writing')
    96→
    97→  // AI provider state
    98→  const [providerSettingsOpen, setProviderSettingsOpen] = useState(false)
    99→  const [aiProvider, setAiProvider] = useState(() => {
   100→    if (typeof window !== 'undefined') return localStorage.getItem('tf_provider') || 'z-ai'
   101→    return 'z-ai'
   102→  })
   103→  const [aiApiKey, setAiApiKey] = useState(() => {
   104→    if (typeof window !== 'undefined') return localStorage.getItem('tf_apiKey') || ''
   105→    return ''
   106→  })
   107→  const [aiBaseUrl, setAiBaseUrl] = useState(() => {
   108→    if (typeof window !== 'undefined') return localStorage.getItem('tf_baseUrl') || ''
   109→    return ''
   110→  })
   111→  const [aiModel, setAiModel] = useState(() => {
   112→    if (typeof window !== 'undefined') return localStorage.getItem('tf_model') || ''
   113→    return ''
   114→  })
   115→
   116→  const saveProviderSettings = useCallback(() => {
   117→    localStorage.setItem('tf_provider', aiProvider)
   118→    localStorage.setItem('tf_apiKey', aiApiKey)
   119→    localStorage.setItem('tf_baseUrl', aiBaseUrl)
   120→    localStorage.setItem('tf_model', aiModel)
   121→    setProviderSettingsOpen(false)
   122→  }, [aiProvider, aiApiKey, aiBaseUrl, aiModel])
   123→
   124→  const clearProviderSettings = useCallback(() => {
   125→    localStorage.removeItem('tf_provider')
   126→    localStorage.removeItem('tf_apiKey')
   127→    localStorage.removeItem('tf_baseUrl')
   128→    localStorage.removeItem('tf_model')
   129→    setAiProvider('z-ai')
   130→    setAiApiKey('')
   131→    setAiBaseUrl('')
   132→    setAiModel('')
   133→  }, [])
   134→
   135→  // Director state
   136→  const [directorLoading, setDirectorLoading] = useState(false)
   137→  const [directorFeedback, setDirectorFeedback] = useState('')
   138→
   139→  // Auto-save
   140→  const saveTimer = useRef<ReturnType<typeof setTimeout> | null>(null)
   141→  const localContent = useRef<string>('')
   142→
   143→  // Load thesis on mount
   144→  useEffect(() => {
   145→    async function load() {
   146→      try {
   147→        // Seed ensures thesis exists
   148→        await fetch('/api/thesis/seed', { method: 'POST' })
   149→        const res = await fetch('/api/thesis')
   150→        const data = await res.json()
   151→        const thesisData = data.thesis || data
   152→        if (thesisData?.id) {
   153→          setThesis(thesisData)
   154→          if (thesisData.chapters?.length > 0) {
   155→            setActiveChapterId(thesisData.chapters[0].id)
   156→          }
   157→        }
   158→      } catch (err) {
   159→        console.error('Failed to load thesis:', err)
   160→      } finally {
   161→        setLoading(false)
   162→      }
   163→    }
   164→    load()
   165→  }, [])
   166→
   167→  const activeChapter = thesis?.chapters.find(c => c.id === activeChapterId)
   168→  const chapterMeta = CHAPTERS.find(c => c.order === activeChapter?.order)
   169→  const colors = chapterMeta ? CHAPTER_COLORS[chapterMeta.color] : CHAPTER_COLORS.emerald
   170→
   171→  // Total word count
   172→  const totalWords = thesis?.chapters.reduce((sum, c) => sum + c.wordCount, 0) || 0
   173→
   174→  // ─── Auto-save handler ───────────────────────────────────
   175→  const handleContentChange = useCallback((content: string) => {
   176→    if (!activeChapter) return
   177→    localContent.current = content
   178→
   179→    // Optimistic update
   180→    setThesis(prev => prev ? {
   181→      ...prev,
   182→      chapters: prev.chapters.map(c =>
   183→        c.id === activeChapterId ? { ...c, content } : c
   184→      ),
   185→    } : null)
   186→
   187→    // Debounced save
   188→    setSaveStatus('idle')
   189→    if (saveTimer.current) clearTimeout(saveTimer.current)
   190→    saveTimer.current = setTimeout(async () => {
   191→ setSaveStatus('saving')
   192→      try {
   193→        const wordCount = content.split(/\s+/).filter(w => w.length > 0).length
   194→        const res = await fetch(`/api/thesis/chapters/${activeChapterId}`, {
   195→          method: 'PATCH',
   196→          headers: { 'Content-Type': 'application/json' },
   197→          body: JSON.stringify({ content, wordCount, status: wordCount > 0 ? 'in_progress' : 'draft' }),
   198→        })
   199→        if (res.ok) {
   200→          const data = await res.json()
   201→          const updated = data.chapter || data
   202→          setThesis(prev => prev ? {
   203→            ...prev,
   204→            chapters: prev.chapters.map(c => c.id === updated.id ? { ...c, ...updated } : c),
   205→          } : null)
   206→          setSaveStatus('saved')
   207→          setTimeout(() => setSaveStatus('idle'), 2000)
   208→        } else {
   209→          setSaveStatus('error')
   210→        }
   211→      } catch {
   212→        setSaveStatus('error')
   213→      }
   214→    }, 2000)
   215→  }, [activeChapter, activeChapterId])
   216→
   217→  // ─── Director submit ─────────────────────────────────────
   218→  const handleDirectorSubmit = useCallback(async () => {
   219→    if (!activeChapter || !chapterMeta) return
   220→    setDirectorLoading(true)
   221→    setDirectorFeedback('')
   222→    try {
   223→      const res = await fetch('/api/directeur', {
   224→        method: 'POST',
   225→        headers: { 'Content-Type': 'application/json' },
   226→        body: JSON.stringify({
   227→          chapitreTitre: `${chapterMeta.number}. ${chapterMeta.title}`,
   228→          chapitreContenu: activeChapter.content,
   229→          probleme: { quoi: 'Contenu du chapitre soumis', comment: 'Évaluation qualitative', pourquoi: 'Validation avant passage au chapitre suivant' },
   230→          hypothese: { texte: 'Chapitre soumis pour évaluation', observation: true, verifiable: true, coherente: true },
   231→          sousDomaineLabel: thesis?.field || 'Non précisé',
   232→          contraintesMethodologiques: '',
   233→        }),
   234→      })
   235→      const data = await res.json()
   236→      if (!res.ok) throw new Error(data.error)
   237→      setDirectorFeedback(data.response)
   238→      // Update chapter status
   239→      await fetch(`/api/thesis/chapters/${activeChapterId}`, {
   240→        method: 'PATCH',
   241→        headers: { 'Content-Type': 'application/json' },
   242→        body: JSON.stringify({ status: 'submitted', directorFeedback: data.response }),
   243→      })
   244→    } catch (err) {
   245→      setDirectorFeedback(err instanceof Error ? err.message : 'Erreur lors de la soumission.')
   246→    } finally {
   247→      setDirectorLoading(false)
   248→    }
   249→  }, [activeChapter, chapterMeta, thesis, activeChapterId])
   250→
   251→  // ─── AI chat ────────────────────────────────────────────
   252→  const handleAiSend = useCallback(async () => {
   253→    if (!aiInput.trim() || aiLoading) return
   254→    const msg = aiInput.trim()
   255→    setAiInput('')
   256→    const newMessages: ChatMsg[] = [...aiMessages, { role: 'user', content: msg }]
   257→    setAiMessages(newMessages)
   258→    setAiLoading(true)
   259→    try {
   260→      const reqBody: Record<string, unknown> = { mode: aiMode, message: msg, temperature: 0.7, maxTokens: 2048, thinking: 'disabled' }
   261→      if (aiProvider !== 'z-ai') {
   262→        reqBody.provider = aiProvider
   263→        reqBody.apiKey = aiApiKey
   264→        reqBody.baseUrl = aiBaseUrl
   265→        reqBody.model = aiModel
   266→      }
   267→      const res = await fetch('/api/ai-writing', {
   268→        method: 'POST',
   269→        headers: { 'Content-Type': 'application/json' },
   270→        body: JSON.stringify(reqBody),
   271→      })
   272→      const data = await res.json()
   273→      if (!res.ok) throw new Error(data.error)
   274→      setAiMessages(prev => [...prev, { role: 'assistant', content: data.response }])
   275→    } catch (err) {
   276→      setAiMessages(prev => [...prev, { role: 'assistant', content: `Erreur : ${err instanceof Error ? err.message : 'inconnue'}` }])
   277→    } finally {
   278→      setAiLoading(false)
   279→    }
   280→  }, [aiInput, aiLoading, aiMessages, aiMode])
   281→
   282→  // ─── Render ─────────────────────────────────────────────
   283→  if (loading) {
   284→    return (
   285→      <div className="h-screen flex items-center justify-center bg-slate-50">
   286→        <div className="text-center space-y-3">
   287→          <Loader2 className="h-8 w-8 text-emerald-500 animate-spin mx-auto" />
   288→          <p className="text-sm text-muted-foreground">Chargement de votre thèse...</p>
   289→        </div>
   290→      </div>
   291→    )
   292→  }
   293→
   294→  if (!thesis) {
   295→    return (
   296→      <div className="h-screen flex items-center justify-center bg-slate-50">
   297→        <div className="text-center space-y-3">
   298→          <AlertTriangle className="h-8 w-8 text-amber-500 mx-auto" />
   299→          <p className="text-sm text-muted-foreground">Impossible de charger la thèse.</p>
   300→          <Button onClick={() => window.location.reload()} variant="outline" size="sm"><RefreshCw className="h-3 w-3 mr-1" />Réessayer</Button>
   301→        </div>
   302→      </div>
   303→    )
   304→  }
   305→
   306→  return (
   307→    <div className="h-screen flex flex-col overflow-hidden bg-slate-50">
   308→      {/* ── MOBILE OVERLAY ── */}
   309→      {sidebarOpen && isMobile && (
   310→        <div className="fixed inset-0 bg-black/50 backdrop-blur-sm z-30" onClick={() => setSidebarOpen(false)} />
   311→      )}
   312→
   313→      <div className="flex flex-1 min-h-0">
   314→        {/* ═══ LEFT SIDEBAR ═══ */}
   315→        <aside className={cn(
   316→          'bg-slate-900 border-r border-slate-800 flex flex-col z-40 shrink-0 transition-all duration-300 shadow-2xl',
   317→          'w-64',
   318→          isMobile && !sidebarOpen && 'fixed -translate-x-full',
   319→          isMobile && sidebarOpen && 'fixed translate-x-0',
   320→          !isMobile && 'relative',
   321→        )}>
   322→          {/* Brand */}
   323→          <div className="flex items-center gap-3 px-3 py-4 mb-2">
   324→            <div className="w-9 h-9 rounded-xl bg-gradient-to-tr from-emerald-600 to-emerald-400 flex items-center justify-center text-white shadow-[0_0_16px_rgba(16,185,129,0.3)] shrink-0">
   325→              <GraduationCap className="h-4 w-4" />
   326→            </div>
   327→            <div className="overflow-hidden">
   328→              <h1 className="font-bold text-sm text-white tracking-tight leading-tight">ThesisFrame</h1>
   329→              <p className="text-[10px] text-emerald-400 font-medium">{thesis.title}</p>
   330→            </div>
   331→            {isMobile && (
   332→              <button onClick={() => setSidebarOpen(false)} className="ml-auto text-slate-400 hover:text-white p-1"><X className="h-4 w-4" /></button>
   333→            )}
   334→          </div>
   335→
   336→          {/* Progress */}
   337→          <div className="px-3 pb-3">
   338→            <div className="flex items-center justify-between mb-1">
   339→              <span className="text-[10px] text-slate-500 font-medium">PROGRESSION</span>
   340→              <span className="text-[10px] text-emerald-400 font-bold">{totalWords.toLocaleString()} mots</span>
   341→            </div>
   342→            <div className="h-1.5 bg-slate-800 rounded-full overflow-hidden">
   343→              <div className="h-full bg-gradient-to-r from-emerald-500 to-emerald-400 rounded-full transition-all duration-500" style={{ width: `${Math.min(100, (totalWords / 80000) * 100)}%` }} />
   344→            </div>
   345→          </div>
   346→
   347→          <Separator className="bg-slate-800" />
   348→
   349→          {/* Chapters list */}
   350→          <div className="px-3 pt-2 pb-1">
   351→            <span className="text-[10px] font-bold uppercase tracking-wider text-slate-500">Chapitres</span>
   352→          </div>
   353→          <nav className="flex-1 overflow-y-auto space-y-0.5 px-2 min-h-0">
   354→            {thesis.chapters.map((ch) => {
   355→              const meta = CHAPTERS.find(m => m.order === ch.order)
   356→              const isActive = ch.id === activeChapterId
   357→              const Icon = (meta?.icon && ICON_MAP[meta.icon]) || FileText
   358→              const chColors = meta ? CHAPTER_COLORS[meta.color] : CHAPTER_COLORS.emerald
   359→              return (
   360→                <button
   361→                  key={ch.id}
   362→                  onClick={() => { setActiveChapterId(ch.id); setSidebarOpen(false) }}
   363→                  className={cn(
   364→                    'w-full p-2.5 flex items-start gap-2.5 rounded-xl text-left transition-all duration-200 group relative overflow-hidden',
   365→                    isActive
   366→                      ? 'bg-gradient-to-r from-emerald-900/80 to-slate-900 text-white border border-emerald-500/40'
   367→                      : 'text-slate-400 hover:text-white hover:bg-slate-800/80 border border-transparent',
   368→                  )}
   369→                >
   370→                  <span className="absolute inset-0 bg-gradient-to-r from-transparent via-emerald-500/10 to-transparent -translate-x-full group-hover:translate-x-full transition-transform duration-700" />
   371→                  <div className={cn('w-7 h-7 rounded-lg flex items-center justify-center shrink-0 mt-0.5 transition-colors',
   372→                    isActive ? chColors.bg.replace('bg-', 'bg-').replace('/500', '/500/20') : 'bg-slate-800 group-hover:bg-slate-700',
   373→                  )}>
   374→                    <Icon className={cn('h-3.5 w-3.5', isActive ? 'text-emerald-400' : 'text-slate-400 group-hover:text-emerald-400')} />
   375→                  </div>
   376→                  <div className="flex-1 min-w-0">
   377→                    <div className="flex items-center gap-1.5">
   378→                      <span className={cn('text-[10px] font-bold', isActive ? 'text-emerald-400' : 'text-slate-600')}>{ch.number}</span>
   379→                      <span className={cn('text-[11px] font-semibold leading-tight truncate', isActive && 'text-white')}>{ch.title}</span>
   380→                    </div>
   381→                    <div className="flex items-center gap-1.5 mt-0.5">
   382→                      <div className={cn('h-1.5 w-1.5 rounded-full', STATUS_COLORS[ch.status] || STATUS_COLORS.draft)} />
   383→                      <span className="text-[9px] text-slate-600">{ch.wordCount.toLocaleString()} mots</span>
   384→                    </div>
   385→                  </div>
   386→                </button>
   387→              )
   388→            })}
   389→          </nav>
   390→
   391→          <Separator className="bg-slate-800" />
   392→
   393→          {/* Tools — always visible, never shrunk */}
   394→          <div className="shrink-0 px-3 pt-2 pb-1">
   395→            <span className="text-[10px] font-bold uppercase tracking-wider text-slate-500">Outils</span>
   396→          </div>
   397→          <div className="shrink-0 px-2 pb-3 space-y-0.5">
   398→            <button onClick={() => setRefsOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   399→              <Library className="h-3.5 w-3.5" /><span>Références biblio.</span>
   400→            </button>
   401→            <button onClick={() => setResourcesOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   402→              <BookOpen className="h-3.5 w-3.5" /><span>Guide rédaction</span>
   403→            </button>
   404→            <button onClick={() => setExportOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   405→              <Download className="h-3.5 w-3.5" /><span>Export PDF</span>
   406→            </button>
   407→            <button onClick={() => setLiteratureOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   408→              <Search className="h-3.5 w-3.5" /><span>Recherche litt.</span>
   409→            </button>
   410→            <button onClick={() => setBalanceOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   411→              <Scale className="h-3.5 w-3.5" /><span>Équilibre chapitres</span>
   412→            </button>
   413→            <button onClick={() => setCloudDriveOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   414→              <Cloud className="h-3.5 w-3.5" /><span>Sauvegarde cloud</span>
   415→            </button>
   416→            <button onClick={() => setJournalFinderOpen(true)} className="w-full p-2 flex items-center gap-2.5 rounded-lg text-slate-400 hover:text-white hover:bg-slate-800/80 transition-all text-xs">
   417→              <Newspaper className="h-3.5 w-3.5" /><span>Journaux OA</span>
   418→            </button>
   419→          </div>
   420→
   421→          {/* User */}
   422→          <div className="shrink-0 p-3 border-t border-slate-800">
   423→            <div className="p-2 rounded-xl bg-slate-800/50 border border-slate-700/40 flex items-center gap-2">
   424→              <div className="w-7 h-7 rounded-full bg-gradient-to-br from-emerald-500 to-emerald-700 flex items-center justify-center text-white font-bold text-[10px] ring-2 ring-emerald-500/30 shrink-0">DR</div>
   425→              <div className="overflow-hidden flex-1">
   426→                <div className="text-[10px] font-medium text-slate-200 truncate">{thesis.author}</div>
   427→                <div className="text-[9px] text-slate-500 truncate">{thesis.university}</div>
   428→              </div>
   429→            </div>
   430→          </div>
   431→        </aside>
   432→
   433→        {/* ═══ MAIN AREA ═══ */}
   434→        <div className="flex-1 flex flex-col min-w-0">
   435→          {/* Chapter header bar */}
   436→          <header className="bg-white/90 backdrop-blur-md border-b border-slate-200 px-4 sm:px-6 py-2.5 flex items-center gap-3 shrink-0 z-20">
   437→            {isMobile && (
   438→              <button onClick={() => setSidebarOpen(true)} className="p-1.5 text-slate-600 hover:text-slate-900 hover:bg-slate-100 rounded-lg">
   439→                <Menu className="h-4 w-4" />
   440→              </button>
   441→            )}
   442→            {chapterMeta && (
   443→              <>
